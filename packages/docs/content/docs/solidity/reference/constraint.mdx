```solidity
struct Constraint {
    /// SCOPE_CONTEXT or SCOPE_CALLDATA.
    uint8 scope;
    /// BE16-encoded path to the target value.
    bytes path;
    /// Encoded operators. Each item = opCode(1) || data.
    bytes[] operators;
}
```
